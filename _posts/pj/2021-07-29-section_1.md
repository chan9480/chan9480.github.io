def to_int(x):
  x=str(x)
  if 'K' in x:
    x=x.replace('K','')
    x=int(x)
  else:
    x=x.replace('M','')
    x=int(float(x)*1000)
  return x

  df['NA_Sales']=df['NA_Sales'].apply(to_int)
  df['EU_Sales']=df['EU_Sales'].apply(to_int)
  df['JP_Sales']=df['JP_Sales'].apply(to_int)
